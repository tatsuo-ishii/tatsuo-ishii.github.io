create table (index varchar, byte bytea);

byte[] byte = "あいうえお".getBytes();
Connection conn = DriverManager.getConnection(url, userID, passwd);

(?,?)");
ps.setString(1, "sample");
ps.setBytes(2, byte);
ps.close();
