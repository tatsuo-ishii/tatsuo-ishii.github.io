<html>
<body>

<?

    echo "$REMOTE_HOST から $HTTP_USER_AGENT を使って、ようこそ。<br>";

    $Hour = Date( "H" );        /* Hours in 24 hour format */
    if ( $Hour < 10 ) {
        echo "おはよう！";
    } elseif ( $Hour < 18 ) {
        echo "こんにちは！";
    } else {
        echo "こんばんは！";
    }


?>

</body>
</html>
