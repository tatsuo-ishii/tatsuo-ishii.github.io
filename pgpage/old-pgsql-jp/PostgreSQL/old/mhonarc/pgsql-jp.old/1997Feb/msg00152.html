幸いなことに、番号での指定でうまく動くようです。

test=> create table mitsu (d date, n int4);
... 値をいれて ...
test=> select * from mitsu order by 2 desc;
         d| n
----------+--
02-01-1997|20
04-01-1997|18
03-01-1997|15
01-01-1997|10
(4 rows)
test=> select * from mitsu order by 1 desc;
         d| n
----------+--
04-01-1997|18
03-01-1997|15
02-01-1997|20
01-01-1997|10
(4 rows)

test=> select d, avg(n) from mitsu group by 1;
         d|avg
----------+---
01-01-1997| 10
02-01-1997| 20
03-01-1997| 15
04-01-1997| 18
(4 rows)
