(provide 'psql)
(defun psql (dbname)
  "Run psql in a comint buffer and switch to it. 
The buffer name is *psql*.  The default database name is given
by the \"USER\" environment variable, or may be specified explicitly.
Any hooks on `comint-psql-hook' is run in the local buffer.
See `make-comint' and `comint-exec'."
  (interactive "sDatabase name: ")
  (let ((program "psql")
        (name "psql")
        (dbname (if (string= dbname "") (getenv "USER") dbname))
        (process-environment (nconc (list "PAGER=" "EDITOR=emacsclient")
                                     process-environment)))
    (switch-to-buffer (make-comint name program nil "-n" dbname))
    (setq comint-process-echoes nil)
    (run-hooks (intern-soft "comint-psql-hook"))))
