#!/bin/sh
PSUSER=pgsql

PATH=${PATH}:/usr/pkg/pgsql/bin
MANPATH=${MANPATH}:/usr/pkg/pgsql/man
PGLIB=/usr/pkg/pgsql/lib
PGDATA=/usr/pkg/pgsql/data
export PATH MANPATH PGLIB PGDATA

cd /usr/pkgsrc/databases/postgresql/work/postgresql-6.5.1/src/test
chown -R $PSUSER regress
cd regress
gmake all 
su $PSUSER -c "gmake runtest"
