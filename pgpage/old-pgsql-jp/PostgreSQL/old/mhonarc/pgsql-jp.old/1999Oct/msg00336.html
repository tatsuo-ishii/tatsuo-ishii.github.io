.cshrcには以下のように書き込まれています。

% cat .cshrc
#csh .cshrc file

alias h         history 25
alias j         jobs -l
alias la        ls -a
alias lf        ls -FA
alias ll        ls -lA

setenv  EDITOR  /usr/bin/ee
setenv  PAGER   more
setenv  BLOCKSIZE       K

setenv PATH "$PATH":/usr/local/pgsql/bin
setenv POSTGRES_HOME /usr/local/pgsql
setenv PGLIB $POSTGRES_HOME/usr/local/pgsql/lib
setenv PGDATA $POSTGRES_HOME/data
setenv MANPATH "$MANPATH":$POSTGRES_HOME/man
setenv LD_LIBRARY_PATH "$LD_LIBRARY_PATH":"$PGLIB"

if ($?MANPATH) then
else
setenv MANPATH
/usr/man:/usr/local/man:/usr/local/apache/man:/usr/share/man:/usr/X11R6/man:
"$POSTGRE
S_HOME"/man
endif


if ($?prompt) then
        # An interactive shell -- set some stuff up
        set filec
        set history = 100
        set savehist = 1000
        set mail = (/var/mail/$USER)
endif
