make distcleanの方法がわからないので下記のようにしました。
cd /usr/local/src/
tar xfz /tmp/postgresql-7.0.tar.gz
cd postgresql-7.0/src
./configure --enable-multibyte=EUC_JP
make distclean
cd /usr/local/src/
tar xfz /tmp/postgresql-7.0.tar.gz
cd postgresql-7.0/src
./configure --enable-multibyte=EUC_JP
make
make install
cd ../doc
make install
initdb --pglib=/usr/local/pgsql/lib
pg_ctl -w start
やはり同じメッセージが出ます。
