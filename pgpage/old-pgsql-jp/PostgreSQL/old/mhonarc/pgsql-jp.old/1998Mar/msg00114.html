>次は mb patch のテストをしてみます。

データベース名、テーブル名、カラム名に日本語を使わなければOKのようです。
テストしたのは以下の SQL 文です。

drop table t2;
create table t2 (a text, b char(16));
insert into t2 values('コンピュータディスプレイ','機A01上');
insert into t2 values('コンピュータ大学','機B01上');
select * from t2;
select * from t2 where a ~ 'コンピュータディスプレイ';
select * from t2 where a ~* 'コンピュータディスプレイ';
select * from t2 where a ~ '大学';
select * from t2 where a ~* '大学';
