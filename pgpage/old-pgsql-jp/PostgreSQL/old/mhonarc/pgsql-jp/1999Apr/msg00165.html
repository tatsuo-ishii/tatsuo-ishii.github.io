ＷＩＮＮＴ４０＋ＯＤＢＣ（６．４）＋ＶＢ５経由で利用しています。
この操作後、いつも,   max(id)＝0  になるような感じです。（ＳＱＬインタープ
リタでも確認）


−−−−  ＶＢ５ のプログラム −−−−−
       'データベースの接続
    Set dao_wsp = CreateWorkspace("seisanbu", "postgres", "*****",
dbUseODBC)
    dao_wsp.DefaultCursorDriver = dbUseODBCCursor
    Set dao_con = dao_wsp.OpenConnection("PostgreSQL", dbDriverComplete,
False,

"ODBC;DSN=PostgreSQL;DATABASE=seisanbu;SERVER=192.*.*.*;
                          PORT=5432;UID=postgres;PWD=*****;READONLY=0;
                          PROTOCOL=;FAKEOIDINDEX=0;SHOWOIDCOLUMN=0;

ROWVERSIONING=0;SHOWSYSTEMTABLES=0;CONNSETTINGS=")

    Send_Str = "select max(id) from mw_bumon"
    Set dao_rst = dao_con.OpenRecordset(Send_Str, dbOpenDynaset)
    If IsNull(dao_rst.Fields(0).Value) Then                        'ここ
で、最大値をチェックする。
        ID_MAX = 1
    Else
        ID_MAX = dao_rst.Fields(0).Value + 1
    End If
