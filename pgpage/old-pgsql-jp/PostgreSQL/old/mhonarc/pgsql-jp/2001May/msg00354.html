自己レスですが，lo_exportの戻り値を利用する必要がないのなら，
SELECTよりもPERFORMを使った方がいいかも知れません．

参考: 付属ドキュメントのplpgsql-description.html

create function spPic_Export(int4,text) returns text as '
declare
begin
	PERFORM lo_export(12940298,''/tmp/pic/pic/12940298'');
	return ''OK'';
end;
' language 'plpgsql';
