こんなかんじでできますね。
---------------------------------------------------------
create table そのいち (こうもく int8,でーた char(10));
insert into そのいち values(1024,'DATA');
ALTER TABLE そのいち RENAME to そのに;
create table そのいち(こうもく char(10),でーた char(10));
insert into そのいち select cast(こうもく as varchar(10)),
でーた from そのに;
\d そのいち
select * from そのいち;
drop table そのに;
---------------------------------------------------------
ALTER TABLEだと属性の変更が利かないんですよね。
