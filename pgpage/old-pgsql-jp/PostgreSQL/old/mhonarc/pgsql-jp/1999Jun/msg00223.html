・環境
  PostgreSQL 6.3.2

  gcc 2.7.2.1
  bison  1.25

・実行方法
Mac からTelnet でサーバにつなげて、psql を実行して以下の
SQL 文をカット＆ペースト

create table tableA (
  code text,
  name text,
  color text,
  istemp bool
);
insert into tableA values (100, 'one', 'white', 'f');
insert into tableA values (200, 'two', 'green', 't');

select * from tableA;

select * from tableA where istemp = 't' ;
select * from tableA where istemp = 'true' ;
select * from tableA where istemp = true ;
