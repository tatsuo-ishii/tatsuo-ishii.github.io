SELECT t_work.workid, t_company.name FROM t_work INNER JOIN t_company ON t_work.companyid = t_company.companyid;

SELECT t_work.workid, t_company.name, t_charged.name FROM (t_work INNER

t_charged ON t_work.chargedid = t_charged.chargedid;
