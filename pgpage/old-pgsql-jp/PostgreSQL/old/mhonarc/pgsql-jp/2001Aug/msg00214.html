<?
 $conn = pg_Connect("localhost", "5432", "", "", "DB名");
 if (!$conn) {
  echo "エラー発生\n";
  exit;
 }

$query = "copy テーブル名 from 'テキストファイル' using delimiters ','";
$query1 = "copy テーブル名 from 'テキストファイル' using delimiters ','";
$query2 = "copy テーブル名 from 'テキストファイル' using delimiters ','";

$ret = pg_Exec($conn,$query);
$ret = pg_Exec($conn,$query1);
$ret = pg_Exec($conn,$query2);


 pg_Close($conn);
?>
