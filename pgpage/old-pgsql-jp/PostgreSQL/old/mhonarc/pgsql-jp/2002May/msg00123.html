#!/usr/local/perl
#--------------------------------------
# 環境設定ここから
#

# データベース名
$dbname = "TEST";

#
# 環境設定ここまで
#--------------------------------------

#$debug_printSQL = 1; # SQLを標準出力に送る
#$debug_noExec   = 1;   # SQL実行しない

        print "dbname=$dbname\n";
}
