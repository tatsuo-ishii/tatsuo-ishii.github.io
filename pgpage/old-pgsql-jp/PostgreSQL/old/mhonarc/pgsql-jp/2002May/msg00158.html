あさかいさま、ありがとうございます。

postgrestと別ユーザーでコンパイルするのに
次のような makefile を作成し、うまくいきました。

INC = -I/usr/local/pgsql/include
CFLAGS =  -O2 -Wall -Wmissing-prototypes -Wmissing-declarations $(INC)
LIBS = -lz -lcrypt -lresolv -lnsl -ldl -lm -lbsd -lreadline -lncurses \
                -L/usr/local/pgsql/lib -lpq

testlibpq : testlibpq.o
        gcc -o $@ testlibpq.o $(LIBS)
