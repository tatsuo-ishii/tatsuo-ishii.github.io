ご希望されている処理は以下のような感じでどうでしょうか？

sub hogehoge( $ )
{
  my( $qstr ) = @_ ;

  my $code = PGRES_COMMAND_OK ;
  $code = PGRES_TUPLES_OK if( $qstr =~ m/^select/i );
  my $result = $conn->exec( $qstr );  # $conn はどこかで取得して下さい
  die $conn->errorMessage unless( $code eq $result->resultStatus );

  return $result ;
}
