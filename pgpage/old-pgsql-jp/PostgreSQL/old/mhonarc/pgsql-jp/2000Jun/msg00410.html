これはperlあたりでscriptかいちゃったほうがてっとりばやいかと
Pg.pmを使うとして、おおざっぱに書くと

----
#!/usr/bin/perl
use Pg;
$conn = Pg::connectdb(接続用文字列);
while (<>) {
	# filedをsplitする
	($name, $section, $age, $telno1, $telno2) = split(/\,/);
	
	# memberにinsert
	$sql = qq(INSERT INTO member ('$name', '$section', '$age'));
	$conn->exec($sql)'
	
	# telephoneにinsert
	$sql = qq(INSERT INTO telphone ('$name', $telno1));
	$conn->exec($sql)'
	
	$sql = qq(INSERT INTO telphone ('$name', $telno2));
	$conn->exec($sql)'
	
}
----
こんな感じですから。。。
