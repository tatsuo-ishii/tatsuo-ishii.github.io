そうですか？ PostgreSQL 6.5.3 で試してみましたが、こちらでは以下のように
なります。

FreeBSD 3.3 で自分で make したもので、漢字コードはすべて EUC でやっています。


  kgotoh=> create table test2 (name varchar, search_name varchar);
  CREATE
  kgotoh=> insert into test2 (name, search_name) values('青木 太郎', 'あおきたろう');
  INSERT 1333290 1
  kgotoh=> insert into test2 (name, search_name) values('鈴木 大介', 'すずきだいすけ');
INSERT 1333291 1
  kgotoh=> select name from test2 where substring(search_name from 1 for 1) between 'あ' and 'お' ;
  name     
  ---------
  青木 太郎
  (1 row)

  kgotoh=> select name from test2 where substring(search_name from 1 for 1) between 'か' and 'こ' ;
  name
  ----
  (0 rows)

  kgotoh=> select name from test2 where substring(search_name from 1 for 1) between 'さ' and 'そ' ;
  name     
  ---------
  鈴木 大介
  (1 row)
