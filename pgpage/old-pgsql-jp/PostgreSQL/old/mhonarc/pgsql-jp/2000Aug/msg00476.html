<?php

function to_unsigned_float($OID)
{
    if ($OID < 0)
        return  $OID + 4294967296; // あふれたなら 補う

    return $OID;
}

function to_signed_int($OID)
{
    if ((float)$OID < 2147483648)
        return $OID;

    return $OID - 4294967296;  // 補った分を引く
}
