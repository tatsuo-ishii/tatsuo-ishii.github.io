> 関数を作ればできます。
> create function firstval(_int4) returns int4 as '
> DECLARE
>   a ALIAS FOR $1;
> BEGIN
>   RETURN a[1];
> END;
> ' LANGUAGE 'plpgsql';
> 
> で
> 
> create index ix1 on t (firstval(code));
