;;; INSERT 38183625 1
;;; postgres=# create view test_view as select f1,f2,f1*f2/100 as f3 from test;

;;; postgres=# select f1,f2,f3,f1+f3 as total from test_view;
;;;  f1  | f2  |         f3         |       total
;;; -----+-----+--------------------+-------------------

;;;  0.7 |   0 |                  0 | 0.699999988079071
;;;  1.5 |   0 |                  0 |               1.5

;;; 
;;;
