  環境は Plamo-1.4.x ベースです。GD1.3 がインストールされている
ので、競合させないように Makefile を一部編集して、スタティック
ライブラリを明示的に指定しました。


APXS_LDFLAGS = /usr/local/src/devel/gd-1.7.3/libgd.a \
	/usr/local/lib/libpng.a /usr/lib/libz.a \
	-L/usr/local/lib -lttf -L/usr/local/pgsql/lib -lpq \
	-lgdbm  pcrelib/libpcre.a -lm -ldl 

LIBS =  -Wl,'-rpath /usr/local/src/devel/php-3.0.12jp-beta4/../gd-1.7.3' \
	-Wl,'-rpath /usr/local/lib' -Wl,'-rpath /usr/local/pgsql/lib'  \
	/usr/local/src/devel/gd-1.7.3/libgd.a \
	/usr/local/lib/libpng.a /usr/lib/libz.a \
	-L/usr/local/lib -lttf -L/usr/local/pgsql/lib -lpq \
	pcrelib/libpcre.a regex/libregex.a -lgdbm -lm -ldl
