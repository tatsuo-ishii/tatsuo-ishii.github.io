単純に引き算で良いのでは。

test=> create table t2 (d1 datetime, d2 datetime);
CREATE
test=> insert into t2 values('1999-10-1','1999-6-1');
INSERT 145578 1
test=> select * from t2;
d1                          |d2                          
----------------------------+----------------------------
Fri Oct 01 00:00:00 1999 JST|Tue Jun 01 00:00:00 1999 JST
(1 row)

test=> select d1 - d2 from t2;
?column?  
----------
@ 122 days
(1 row)
