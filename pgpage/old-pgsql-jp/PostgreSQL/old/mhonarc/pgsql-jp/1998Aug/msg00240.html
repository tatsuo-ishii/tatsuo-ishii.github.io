PostgreSQL の配列型にこだわらなければ、普通の SQL でかけますが。
利点は、検索はサーバ任せになることと、
趣味の種類が少々増えても平気だというところです。

create table relation_person_hobby (id_person int4, id_hobby int4) ;

--	1)
--	5番のユーザが 選択肢から 7,8,9 の趣味を選択して格納
-- 初期化
delete from relation_person_hobby where id_person = 5 ;
-- 格納
insert into relation_person_hobby values (5 , 7) ;
insert into relation_person_hobby values (5 , 8) ;
insert into relation_person_hobby values (5 , 9) ;

--	2)
--	趣味が 7 のユーザの一覧
select id_person from relation_person_hobby where id_hobby = 7 ;

--	3)
--	各ユーザの趣味の数
select id_person, count(*) as count_hobby from relation_person_hobby group
by id_person ;

--	4)
--	各趣味のユーザの数
select id_hobby, count(*) as count_person from relation_person_hobby group
by id_hobby ;

----------
