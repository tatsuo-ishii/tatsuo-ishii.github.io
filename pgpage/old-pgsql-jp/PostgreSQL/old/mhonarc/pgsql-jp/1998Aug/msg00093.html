#!/usr/local/bin/perl
#------------------------------------------------------------#
# @(#) jgrep 1.4 15:18:19 95/11/10   Y. Ichikawa (OU/FS/DIS)
#------------------------------------------------------------#
sub usage {
    print STDERR "Usage: $0 pattern [<file1> <file2> ...]\n";
}
sub usage_and_die { usage(); exit 1; }

sub open_file {
    my ($file, $pattern)=@_;
    if (! open(input, "nkf -ue $file |")) {
	printf STDERR "$0: cannot open $file.\n";
	return 0; # false
    }
    return 1; # true
}

$#ARGV < 0 && usage_and_die();
$pr_name = $#ARGV >= 2;
$pattern=shift(@ARGV);
@files=@ARGV;
if ($#files == -1) { @files=('-'); }

LOOP: foreach $file (@files) {
    open_file($file,$pattern) || next LOOP;
    while (<input>) {
	/$pattern/ &&
	    ( print (($pr_name ? "$file:" : ""), "$_"));
    }
}
