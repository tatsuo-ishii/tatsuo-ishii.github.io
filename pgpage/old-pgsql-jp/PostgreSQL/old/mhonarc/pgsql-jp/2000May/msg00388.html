> になったんですが、プラットフォームが HP-UX の WS を使わなければならなく

私は HP-UX 10.20 上で PostgreSQL 6.5.3 を動作し、
以下のスクリプトで起動しています。

#!/bin/sh

POSTGRESDIR=/usr/local/pgsql; export POSTGRESDIR

case "$1" in
'start')
        echo "Starting PostgreSQL postmaster ..."

    if [ -x ${POSTGRESDIR}/bin/postmaster -a -d ${POSTGRESDIR}/data ]; then
        rm -f /tmp/.s.PGSQL.5432
        /usr/bin/su - postgres -c "/usr/local/pgsql/bin/postmaster -B 2048
-S -i -o '-S 2048 -F'"
    fi

        echo "... done."
        ;;

'stop')
        ;;
esac

----------------

> su - postgres -c "/usr/local/pgsql/bin/postmaster \
> -D /usr/local/pgsql/data \
> -i -B 256 -S"
