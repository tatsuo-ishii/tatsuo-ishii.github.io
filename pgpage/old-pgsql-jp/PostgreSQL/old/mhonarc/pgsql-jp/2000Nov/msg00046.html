> 現在DBに登録した英数字をLIKE検索することを考えているんですが、
> 大文字小文字の混在するデータを検索する時に、ユーザーが入力した
> 文字列が小文字のみであっても、大文字の混ざった文字列にヒット
> させることは可能でしょうか？

postgres=> create table test ( name varchar(32) );
CREATE
postgres=> insert into test values ('I Like Milk.');
INSERT 585961 1
postgres=> select name from test where lower(name) like lower('%LikE%');
name
------------
I Like Milk.
(1 row)
postgres=> drop table test;
DROP
