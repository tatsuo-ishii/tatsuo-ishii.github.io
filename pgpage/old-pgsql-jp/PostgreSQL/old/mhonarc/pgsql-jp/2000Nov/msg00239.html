こんなんでもよいです。

tmp=> drop table test;
DROP
tmp=> create table test(
    userid int2,        // ユーザID
    startdate date,     // 開始日
    enddate date        // 終了日
);
tmp-> tmp-> tmp-> tmp-> CREATE
tmp=> insert into test values(1, '2000/11/16', '');
ERROR:  Bad date external representation ''
tmp=> insert into test values(1, '2000/11/16', NULL);
INSERT 1425835 1
tmp=> select * from test;
userid| startdate|enddate
------+----------+-------
     1|11-16-2000|
