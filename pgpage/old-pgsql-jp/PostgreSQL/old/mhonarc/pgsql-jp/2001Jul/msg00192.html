# cd php-4.0.5/ext

# tar xvzf php4_mbregex-1.2.1.tar.gz
# cd ..
# ./buildconf
# ./configure --with-apxs=/usr/local/apache/bin/apxs \
--enable-jstring --enable-mbregex \
--with-pgsql --enable-versioning
# make all
