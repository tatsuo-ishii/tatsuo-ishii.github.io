  • From: Kounosuke Shiomi <yukinopo@livedoor.com>

塩見と申します。

PostgreSQL 7.1.3 にて、alter tableでserial型を追加した時に、うまく
追加できないようです。

# create table test (item text);
# alter table test add column seq serial;
# insert into test (item) values ('test');
# select * from test;
 item | seq
------+-----
 test |
(1 row)


# drop table test;
DROP

# create table test (item text,seq serial);
NOTICE:  CREATE TABLE will create implicit sequence 'test_seq_seq' for SERIAL column 'test.seq'
NOTICE:  CREATE TABLE/UNIQUE will create implicit index 'test_seq_key' for table 'test'
CREATE
# insert into test ( item) values ('test');
# select * from test;

 item | seq
------+-----
 test |   1
(1 row)
