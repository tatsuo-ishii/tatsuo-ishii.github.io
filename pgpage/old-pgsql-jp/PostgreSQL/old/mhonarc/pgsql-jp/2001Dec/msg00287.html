こちらで試してみましたが、ちゃんと書き込まれています。
# こちらのユーザはpgsql です。

$ pg_ctl --version
pg_ctl (PostgreSQL) 7.1.3

$ pwd
/usr/local/pgsql

$ ls logfile.*
ls: No match.

$ ls -ld ../pgsql
drwxr-xr-x  3 pgsql  pgsql  512 Dec 18 19:08 ../pgsql

$ pg_ctl -w start > logfile.1
$ pg_ctl stop

$ su -l pgsql -c "pg_ctl -w start > logfile.2"
$ pg_ctl stop

$ su -l pgsql -c "pg_ctl -w start -l logfile.3"
$ pg_ctl stop

$ ls logfile.*
-rw-r--r--  1 pgsql  pgsql  461 Dec 18 19:09 logfile.1
-rw-r--r--  1 pgsql  pgsql  461 Dec 18 19:11 logfile.2
-rw-------  1 pgsql  pgsql  389 Dec 18 19:11 logfile.3
