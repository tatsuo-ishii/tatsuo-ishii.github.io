お世話様です。
下記の通り変更することで、データーの更新が出来るようになりました。
有り難うございました。

>dbUseODBCCursor -> dbUseODBC に変更
>ws.DefaultCursorDriver = dbUseODBCCursorを追加してください。
>

Dim ws As Workspace
Dim cn As Connection
Dim rs As Recordset
    Set ws = CreateWorkspace("ODBC_WS", "", "", dbUseODBC)
    ws.DefaultCursorDriver = dbUseODBCCursor
    Set cn = ws.OpenConnection("Test_db", dbDriverComplete, False)
    Set rs = cn.OpenRecordset("test_tb", dbOpenDynamic, dbExecDirect, 
dbOptimistic)
