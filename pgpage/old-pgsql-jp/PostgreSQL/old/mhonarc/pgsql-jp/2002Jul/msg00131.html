フォームをロードした時下記のようなプログラムを実行しています。
MS-ACCESSに同じDBを作り連結した時はこのやり方で編集可能な状態になりました。
hdoというのはグリッドの名前です。

グリッドで入力出来なければMS-ACCESSの表形式フォームのような
連続してデータを入力できるフォームを作りたいと思っています。

よろしくお願いいたします。

Dim cn As New ADODB.Connection
Dim rs As New ADODB.Recordset
Dim sql As String

sql = "SELECT No,日付,t_hdo.商品コード, 商品名,規格,出庫数 " _
    & "FROM t_hdo INNER JOIN t_ms2 " _
    & "ON t_hdo.商品コード = t_ms2.商品コード " _
    & "ORDER BY No"

rs.Open sql, cn, adOpenStatic, adLockOptimistic

Set hdo.DataSource = rs
hdo.AllowAddNew = True
hdo.AllowUpdate = True
hdo.AllowDelete = True
hdo.AllowArrows = True

hdo.Columns("日付").Width = 70 * 15
hdo.Columns("商品コード").Width = 60 * 15
hdo.Columns("商品名").Width = 100 * 15
hdo.Columns("規格").Width = 100 * 15
hdo.Columns("出庫数").Width = 60 * 15
