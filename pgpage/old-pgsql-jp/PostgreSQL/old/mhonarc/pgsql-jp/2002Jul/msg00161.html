$sql = "select id from tbl
 where ((keyword1 ILIKE '%".$searchword[0]."%')
 or (keyword2 ILIKE '%".$searchword[0]."%')
 or (keyword3 ILIKE '%".$searchword[0]."%') or ・・・))";
for ($i=1;$1 < sizeof($searchword);$i++){
