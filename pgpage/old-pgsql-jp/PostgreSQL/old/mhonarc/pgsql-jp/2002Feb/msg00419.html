> シリアルのraitenidが時系列の正しさを保証するなら
> 
> select kokyakuid, namae, hizuke, tenpomei
> from (
>         select max(raitenid) as raitenid
>          from raiten_table
>          group by kokyakuid
> ) R0
> join raiten_table R using(raitenid)
> join kokyaku_table K using(kokyakuid)
> ;
