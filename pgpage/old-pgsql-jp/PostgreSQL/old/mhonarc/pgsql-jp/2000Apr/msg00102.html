Kaori Inaba さんは書きました:
>
>いなばといいます。こんにちは。
>
>> aaa=> select * from bbb;
>> more: not found
>> aaa=>
>> 
>> となってしまいます。
>
>これは、 more コマンドがないよ、と言っているのでしょう。
>
>% whereis more 
>
>として出て来た path が 環境変数 PATH に含まれているでしょうか？
>もし、含まれていないなら、環境変数に more コマンドの path を
>追加すればよいでしょう。
>もし、 more コマンドがないなら(ってそんなこと、あるのでしょうか？)
>環境変数 PAGER を他のもの(例えば less とか)に設定すれば良いです。

以下のようになっているので、more コマンドは存在しますし、
path は大丈夫だと思うのですが・・・

% whereis more
more: /usr/bin/more /usr/share/man/man1/more.1.gz
% cat .cshrc
set path        = (/bin /usr/bin /usr/local/bin ~/bin ~/usr/bin 
~/usr/local/bin)

if (! $?prompt) exit

set history     = 100
set prompt      = "`hostname -s`% "

unalias *

alias h         history
alias ll        ls -lg

setenv BLOCKSIZE        1k
setenv EDITOR           vi
setenv PAGER            more

umask 022

if (! $?term) set term = vt100

stty echoe erase ^H kill ^U intr ^C quit ^\| susp ^Z

setenv PGDATA /usr/local/pgsql/data
setenv PGLIB /usr/local/pgsql/lib
setenv LD_LIBRARY_PATH /usr/local/pgsql/lib
set path = (/usr/local/pgsql/bin $path)

%
