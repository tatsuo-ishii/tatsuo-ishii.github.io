# 後は正規表現を使わなければ strstr() でいけそうなんですが...

======================== namelist.sql ===========================
CREATE TYPE namelist (
   internallength = variable, 
   input = array_in,
   output = array_out,
   element = text
);

CREATE FUNCTION namelist_match(namelist, text)
   RETURNS bool
   AS 'namelist.so'
   LANGUAGE 'c';

CREATE OPERATOR ~~ (
   leftarg = namelist, rightarg = text, procedure = namelist_match
)
