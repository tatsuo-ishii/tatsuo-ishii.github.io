CREATE TABLE hoge ( id int4 );
COMMENT ON TABLE  hoge     IS 'ID管理テーブル';
COMMENT ON COLUMN hoge.id  IS '利用者ID';

詳細情報のコメント欄に'ID管理テーブル'が、
フィールド定義の備考欄に'利用者ID'が入ります。
