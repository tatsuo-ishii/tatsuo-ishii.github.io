COUNT(index) に名前を付けて、

String query = "SELECT COUNT(index) as c FROM tests;"
Connection conn = DriverManager.getConnection(url, account, password);
Statement st = conn.createStatement();
ResultSet rs = st.executeQuery(query);
while(rs.next()) {
  total = rs.getInt("c");
}
rs.close();
st.close();
