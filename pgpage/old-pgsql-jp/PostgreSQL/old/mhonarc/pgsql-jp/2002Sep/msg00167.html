tableと同名のtemp tableっておもしろい挙動するんですね。さっき試していて
初めて知りました。使いどころはあまり無いと思いますが。

create table a ... ;
insert into a ... ;
select count(*) from a; -> 1件
create temp table a ... ; 
select count(*) from a; -> 0件
insert into a ...; 
insert into a ...;
select count(*) from a; -> 2件
drop table a;
select count(*) from a; -> 1件
