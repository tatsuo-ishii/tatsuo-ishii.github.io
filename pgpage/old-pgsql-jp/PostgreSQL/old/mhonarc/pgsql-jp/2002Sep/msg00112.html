こんな感じでしょうか。

  test=# \d t
                   Table "t"
   Column |         Type         | Modifiers 
  --------+----------------------+-----------
   a      | character varying(4) | 

  test=# select * from t;
    a   
  ------
   0001
   001
  (2 rows)

  test=# select * from t where to_number(a, '9999') = 1;
    a   
  ------
   0001
   001
  (2 rows)
