create table t7(
  a varchar(04)
);

insert into t7 values('001');
insert into t7 values('0001');
insert into t7 values('002');
insert into t7 values('0002');
insert into t7 values('003');
insert into t7 values('0003');

select to_number(a, '9999') as b from t7 where to_number(a, '9999') = 1;

drop table t7;
