                 TCL_LIBDIR= /usr/local/lib

                 TK_LIBDIR= /usr/local/lib
                 TK_LIB= -ltk4.2jp
                 X11_INCDIR= /usr/include
                 X11_LIBDIR= /usr/X11/lib
                 X11_LIB= -lX11 -lm -lc
