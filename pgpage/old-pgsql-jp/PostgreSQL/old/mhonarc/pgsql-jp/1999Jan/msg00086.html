void unsetenv(const char *name)
{
   int i, l, pos = 0;
   extern char **environ;

   if(!name)
     return;
   l = strlen(name);
   while(environ[pos]){
     if(!strncmp(name, environ[pos], l) && environ[pos][l] == '='){
       for(i=pos; environ[i]; i++)
         environ[i] = environ[i + 1];
       continue;
     }
     pos ++;
   }
}
