表名等は[pgsql-jp 23408]時点の状態から推測した物です。

SELECT * FROM syouhin_master WHERE syouhin_id IN (
 SELECT M.syouhin_id FROM syouhin_master AS M
  LEFT OUTER JOIN siire  S ON (M.syouhin_id=S.syouhin_id)
  LEFT OUTER JOIN uriage U ON (M.syouhin_id=U.syouhin_id)
 GROUP BY M.syouhin_id
 HAVING (SUM(S.kosuu) IS NOT NULL)
    AND (((SUM(S.kosuu)-SUM(U.kosuu))>0)
       OR (SUM(U.kosuu) IS NULL)))
ORDER BY syouhin_id
