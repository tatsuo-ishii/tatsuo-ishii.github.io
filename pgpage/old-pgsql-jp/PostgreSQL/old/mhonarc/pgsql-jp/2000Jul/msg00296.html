思いつきます。
# print 文の辺りが怪しいですが...

# 0行を返すSQL
$sql = "select * from hoge where 1 = 0";
$result = $conn->exec ($sql);

# カラム数を返す
$number_of_clms = $result->nfields;
print "カラム総数は ".$number_of_clms."です。\n"

# カラム名を返す
for ( $i=0; $i < $number_of_clms; $i++) {
	$name_of_clm = $result->fname($i)
	print "$name_of_clm \n"; 
}
