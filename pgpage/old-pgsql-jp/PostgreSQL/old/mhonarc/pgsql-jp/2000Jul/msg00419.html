>tech=# select * from tr_report where mcname LIKE ('CP-2', 'CP-4%');
>ERROR:  parser: parse error at or near ";"

select * from tr_report where mcname IN 'CP-2' OR mcname IN 'CP-4%';
select * from tr_report where mcname ~* 'CP-2' OR mcname ~* 'CP-4%';
select * from tr_report where mcname LIKE 'CP-2' OR mcname LIKE 'CP-4%';
