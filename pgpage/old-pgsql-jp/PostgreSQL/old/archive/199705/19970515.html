			| ((u_char *) &n)[1])
#define ntoh_l(n) (u_long)(((u_char *) &n)[0] << 24 \
			| ((u_char *) &n)[1]\
			 << 16 | ((u_char *)\
			 &n)[2] << 8 | ((u_char *) &n)[3])
#define hton_s(n) (ntoh_s(n))
#define hton_l(n) (ntoh_l(n))

を使うとほとんど(create_* ,copy)以外は、failed.

#defined BYTE_ORDER == LITTLE_ENDIANを明示して

#if BYTE_ORDER == LITTLE_ENDIAN
#  define ntoh_s(n) n
#  define ntoh_l(n) n
#  define hton_s(n) n
#  define hton_l(n) n

を使うと数個のテスト以外はok.
